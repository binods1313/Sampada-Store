## 2️⃣ BLUEPRINT MAPPING ENGINE
### File: `lib/printifyClient.js`

```js
const BLUEPRINT_CACHE = new Map();

export async function getBlueprintSpecs(blueprintId) {
  if (BLUEPRINT_CACHE.has(blueprintId)) return BLUEPRINT_CACHE.get(blueprintId);
  const res = await fetch(`https://api.printify.com/v1/catalog/blueprints/${blueprintId}.json`, {
    headers: { Authorization: `Bearer ${process.env.PRINTIFY_API_KEY}` }
  });
  const data = await res.json();
  BLUEPRINT_CACHE.set(blueprintId, data);
  return data;
}

// Canonical dimension map — extend as you add product types
export const BLUEPRINT_DIMENSION_MAP = {
  // T-shirts (blueprint 5 = Bella Canvas 3001)
  5:   { front: { w: 4500, h: 5400, x: 0, y: 0 }, back: { w: 4500, h: 5400, x: 0, y: 0 } },
  // Mugs (blueprint 384)
  384: { wrap: { w: 2700, h: 1125, x: 0, y: 0 } },
  // Hoodies (blueprint 9)
  9:   { front: { w: 4500, h: 5400, x: 0, y: 300 }, back: { w: 4500, h: 5400, x: 0, y: 200 } },
  // Tote bags (blueprint 77)
  77:  { front: { w: 3600, h: 4200, x: 0, y: 0 } },
  // Phone cases (blueprint 231)
  231: { front: { w: 1200, h: 2208, x: 0, y: 0 } },
  // Posters (blueprint 182)
  182: { front: { w: 4800, h: 6000, x: 0, y: 0 } },
  // ALL-OVER shirts (blueprint 1006)
  1006: { allOver: { w: 5400, h: 7200, x: 0, y: 0 } }
};

export async function autoFillDimensions(blueprintId, placement) {
  // Try static map first (fast), fall back to live API
  const staticDims = BLUEPRINT_DIMENSION_MAP[blueprintId]?.[placement];
  if (staticDims) return staticDims;
  const apiData = await getBlueprintSpecs(blueprintId);
  // Parse print_areas from Printify API response
  const area = apiData.print_areas?.find(a => a.position === placement);
  return area ? { w: area.width, h: area.height, x: area.x || 0, y: area.y || 0 } : null;
}
```

```js
export async function generateMockupWithBranding(productType, baseImageUrl, brandConfig) {
  // brandConfig = { logoUrl, primaryColor, fontFamily, styleGuide }
  const prompt = `
    Generate a professional product mockup for a ${productType}.
    Apply the Sampada brand: logo at ${brandConfig.logoPosition || 'center chest'},
    primary color ${brandConfig.primaryColor}, style: ${brandConfig.styleGuide}.
    Clean white background. Photo-realistic. High contrast.
    Base design: ${baseImageUrl}
  `;
  const res = await fetch('https://generativelanguage.googleapis.com/v1beta/models/imagen-3.0-generate-001:predict', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', 'x-goog-api-key': process.env.GOOGLE_AI_KEY },
    body: JSON.stringify({ instances: [{ prompt }], parameters: { sampleCount: 3 } })
  });
  const data = await res.json();
  return data.predictions; // returns 3 alternates for editor to pick
}
```

### 3b. Watermark / Logo Remover (Google Vision + inpainting)
```js
export async function removeLogo(imageUrl) {
  // Step 1: detect logo bounding boxes via Vision API
  const visionRes = await fetch(`https://vision.googleapis.com/v1/images:annotate?key=${process.env.GOOGLE_VISION_KEY}`, {
    method: 'POST',
    body: JSON.stringify({ requests: [{ image: { source: { imageUri: imageUrl } }, features: [{ type: 'LOGO_DETECTION' }] }] })
  });
  const { responses } = await visionRes.json();
  const logos = responses[0]?.logoAnnotations || [];
  if (!logos.length) return imageUrl; // no logos found, return original

  // Step 2: inpaint detected regions (use Stability AI or Clipdrop)
  const cleanImageUrl = await inpaintRegions(imageUrl, logos.map(l => l.boundingPoly));
  return cleanImageUrl;
}
```

```js
export async function calculateDynamicPrice({ baseCost, demandScore, trendingScore, competitorPrices, targetMargin }) {
```

```js
export default async function handler(req, res) {

  const { _id, _type, printifyIntegration } = req.body;

  try {
    // Auto-pipeline on every new product save
    await runFullPipeline(_id, req.body);
    res.status(200).json({ ok: true });
  } catch (err) {
    // Self-heal: increment retry count, set status to 'retrying'
    await client.patch(_id).set({

    }).commit();
    res.status(500).json({ error: err.message });
  }
}
```

```js
export async function syncToPrintify(sanityDoc) {
  const { printifyIntegration, aiGeneration, revenueConfig } = sanityDoc;
  const body = {
    title: sanityDoc.title,
    description: sanityDoc.aiContent?.autoDescription || sanityDoc.description,
    blueprint_id: printifyIntegration.printifyBlueprintId,
    print_provider_id: printifyIntegration.printProviderId,
    variants: await getPrintifyVariants(printifyIntegration.printifyBlueprintId, printifyIntegration.printProviderId),
    print_areas: [{
      variant_ids: 'all',
      placeholders: [{
        position: printifyIntegration.designPlacement,
        images: [{
          id: await uploadImageToPrintify(printifyIntegration.logoRemovedImage || aiGeneration.autoGeneratedMockup),
          x: printifyIntegration.designX / printifyIntegration.designWidth,
          y: printifyIntegration.designY / printifyIntegration.designHeight,
          scale: printifyIntegration.scale || 1,
          angle: printifyIntegration.angle || 0
        }]
      }]
    }]
  };

  const method = printifyIntegration.printifyProductId ? 'PUT' : 'POST';
  const url = printifyIntegration.printifyProductId
    ? `https://api.printify.com/v1/shops/${process.env.PRINTIFY_SHOP_ID}/products/${printifyIntegration.printifyProductId}.json`
    : `https://api.printify.com/v1/shops/${process.env.PRINTIFY_SHOP_ID}/products.json`;

  const res = await fetch(url, { method, headers: { Authorization: `Bearer ${process.env.PRINTIFY_API_KEY}`, 'Content-Type': 'application/json' }, body: JSON.stringify(body) });
  const data = await res.json();

  await client.patch(sanityDoc._id).set({
    'printifyIntegration.printifyProductId': data.id,
    'printifyIntegration.syncStatus': 'synced',
    'printifyIntegration.lastSyncedAt': new Date().toISOString(),
    'printifyIntegration.retryCount': 0
  }).commit();

  return data;
}
```
